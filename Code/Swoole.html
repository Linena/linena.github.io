    layer.msg("content is ok!");
    return false;
};
//接收消息
webSocket.onmessage = function(event){
    console.log(event.data);
};
//关闭
webSocket.onclose = function(event){
    layer.msg("关闭连接!");
    return false;
};
//错误信息
webSocket.onerror = function (event) {
    layer.msg("网络错误!");
    return false;
};
